# Plan 01 — Skeleton + Auth — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bootstrap the K8s PaaS project. Deliver a runnable Docker Compose stack (Go API + Postgres + React frontend served by nginx) where a user can register, log in, log out, and land on an empty dashboard. No cluster integration yet — that is Plan 02.

**Architecture:** Go API (chi + sqlc + golang-migrate) talks to Postgres. React SPA (Vite + TanStack Router + TanStack Query + shadcn/ui) is served by nginx and proxies `/api` to the Go service. All three components run as containers in a single `docker-compose.yml`.

**Tech Stack:** Go 1.22+, chi, sqlc, golang-migrate, gorilla/sessions, bcrypt, Postgres 16, React 18, TypeScript, Vite, TanStack Router, TanStack Query, Tailwind, shadcn/ui, nginx.

**Responsibilities (one sentence each):**

- `cmd/api/main.go` — process entrypoint; reads env, wires dependencies, starts HTTP server.
- `internal/web` — HTTP layer: router, middleware, JSON helpers, healthz.
- `internal/auth` — register / login / logout handlers; session middleware; password verification flow.
- `internal/crypto/password.go` — bcrypt wrapper.
- `internal/db` — Postgres connection, migration runner, sqlc-generated typed queries.
- `internal/testutil` — testcontainers Postgres helper shared across integration tests.
- `frontend/src/routes/*` — TanStack Router route files; one file per page.
- `frontend/src/api/client.ts` — single source for HTTP calls to the Go API.
- `frontend/src/hooks/useCurrentUser.ts` — exposes the current user via TanStack Query; drives protected routes.
- `nginx.conf` — serves the SPA static bundle and proxies `/api` to the Go service.

import (

func StartPostgres(t *testing.T) string {
	t.Helper()
	ctx := context.Background()

	pg, err := postgres.RunContainer(ctx,
		postgres.WithDatabase("paas"),
		postgres.WithUsername("paas"),
		postgres.WithPassword("paas"),
		postgres.BasicWaitStrategies(),
		postgres.WithSQLDriver("pgx"),
	)
	if err != nil {
		t.Fatalf("start postgres: %v", err)
	}
	t.Cleanup(func() {
		_ = pg.Terminate(context.Background())
	})

	dsn, err := pg.ConnectionString(ctx, "sslmode=disable")
	if err != nil {
		t.Fatalf("dsn: %v", err)
	}

	// Wait for accept connections
	_ = wait.ForListeningPort("5432/tcp").WithStartupTimeout(20 * time.Second)

}

package db

import (
	"fmt"

	"github.com/golang-migrate/migrate/v4"
	_ "github.com/golang-migrate/migrate/v4/database/postgres"
	_ "github.com/golang-migrate/migrate/v4/source/file"
)

func RunMigrations(dsn, sourceDir string) error {
	m, err := migrate.New("file://"+sourceDir, dsn)
	if err != nil {
		return fmt.Errorf("migrate.New: %w", err)
	}
	defer m.Close()

	if err := m.Up(); err != nil && err != migrate.ErrNoChange {
		return fmt.Errorf("migrate.Up: %w", err)
	}
	return nil
}
```

package crypto

import "golang.org/x/crypto/bcrypt"

const bcryptCost = 12

func HashPassword(plain string) (string, error) {
	b, err := bcrypt.GenerateFromPassword([]byte(plain), bcryptCost)
	if err != nil {
		return "", err
	}
	return string(b), nil
}

func CheckPassword(hash, plain string) error {
	return bcrypt.CompareHashAndPassword([]byte(hash), []byte(plain))
}
```

package auth

import (
	"context"
	"crypto/rand"
	"crypto/sha256"
	"encoding/hex"
	"errors"
	"fmt"
	"strings"
	"time"

	"github.com/jackc/pgx/v5/pgtype"
	"github.com/your-org/k8s-paas/api/internal/crypto"
	"github.com/your-org/k8s-paas/api/internal/db"
)

var (
	ErrInvalidCredentials = errors.New("invalid credentials")
	ErrEmailTaken         = errors.New("email already registered")
	ErrSessionInvalid     = errors.New("session invalid or expired")
)

type Service struct {
	q          db.Querier
	SessionTTL time.Duration
}

func NewService(q db.Querier) *Service {
	return &Service{q: q, SessionTTL: 30 * 24 * time.Hour}
}

func (s *Service) Register(ctx context.Context, email, password string) (db.User, string, error) {
	email = strings.ToLower(strings.TrimSpace(email))
	if email == "" || len(password) < 8 {
		return db.User{}, "", fmt.Errorf("email required and password must be 8+ chars")
	}

	count, err := s.q.CountUsers(ctx)
	if err != nil {
		return db.User{}, "", fmt.Errorf("count users: %w", err)
	}

	hash, err := crypto.HashPassword(password)
	if err != nil {
		return db.User{}, "", err
	}

	user, err := s.q.CreateUser(ctx, db.CreateUserParams{
		Email:        email,
		PasswordHash: hash,
		IsAdmin:      count == 0,
	})
	if err != nil {
		if strings.Contains(err.Error(), "duplicate key") {
			return db.User{}, "", ErrEmailTaken
		}
		return db.User{}, "", fmt.Errorf("create user: %w", err)
	}

	token, err := s.issueSession(ctx, user.ID)
	if err != nil {
		return db.User{}, "", err
	}
	return user, token, nil
}

func (s *Service) Login(ctx context.Context, email, password string) (db.User, string, error) {
	email = strings.ToLower(strings.TrimSpace(email))
	user, err := s.q.GetUserByEmail(ctx, email)
	if err != nil {
		return db.User{}, "", ErrInvalidCredentials
	}
	if err := crypto.CheckPassword(user.PasswordHash, password); err != nil {
		return db.User{}, "", ErrInvalidCredentials
	}

	token, err := s.issueSession(ctx, user.ID)
	if err != nil {
		return db.User{}, "", err
	}
	return user, token, nil
}

func (s *Service) ValidateToken(ctx context.Context, token string) (db.User, error) {
	sess, err := s.q.GetSessionByTokenHash(ctx, hashToken(token))
	if err != nil {
		return db.User{}, ErrSessionInvalid
	}
	return s.q.GetUserByID(ctx, sess.UserID)
}

func (s *Service) Logout(ctx context.Context, token string) error {
	return s.q.DeleteSessionByTokenHash(ctx, hashToken(token))
}

func (s *Service) issueSession(ctx context.Context, userID pgtype.UUID) (string, error) {
	raw := make([]byte, 32)
	if _, err := rand.Read(raw); err != nil {
		return "", fmt.Errorf("rand: %w", err)
	}
	token := hex.EncodeToString(raw)

	expires := pgtype.Timestamptz{Time: time.Now().Add(s.SessionTTL), Valid: true}
	if _, err := s.q.CreateSession(ctx, db.CreateSessionParams{
		UserID:     userID,
		TokenHash:  hashToken(token),
		ExpiresAt:  expires,
	}); err != nil {
		return "", fmt.Errorf("create session: %w", err)
	}
	return token, nil
}

func hashToken(token string) string {
	sum := sha256.Sum256([]byte(token))
	return hex.EncodeToString(sum[:])
}

package auth

import (
	"errors"
	"net/http"
	"time"

	"github.com/your-org/k8s-paas/api/internal/web"
)

const cookieName = "session"

type Handlers struct {
	svc *Service
}

func NewHandlers(svc *Service) *Handlers {
	return &Handlers{svc: svc}
}

type credsRequest struct {
	Email    string `json:"email"`
	Password string `json:"password"`
}

type userResponse struct {
	ID      string `json:"id"`
	Email   string `json:"email"`
	IsAdmin bool   `json:"is_admin"`
}

func (h *Handlers) Register(w http.ResponseWriter, r *http.Request) {
	var req credsRequest
	if err := web.DecodeJSON(r, &req); err != nil {
		web.WriteError(w, http.StatusBadRequest, "invalid json")
		return
	}
	user, token, err := h.svc.Register(r.Context(), req.Email, req.Password)
	switch {
	case errors.Is(err, ErrEmailTaken):
		web.WriteError(w, http.StatusConflict, "email already registered")
		return
	case err != nil:
		web.WriteError(w, http.StatusBadRequest, err.Error())
		return
	}

	setSessionCookie(w, token, h.svc.SessionTTL)
	web.WriteJSON(w, http.StatusCreated, userResponse{
		ID: user.ID.String(), Email: user.Email, IsAdmin: user.IsAdmin,
	})
}

func (h *Handlers) Login(w http.ResponseWriter, r *http.Request) {
	var req credsRequest
	if err := web.DecodeJSON(r, &req); err != nil {
		web.WriteError(w, http.StatusBadRequest, "invalid json")
		return
	}
	user, token, err := h.svc.Login(r.Context(), req.Email, req.Password)
	if err != nil {
		web.WriteError(w, http.StatusUnauthorized, "invalid credentials")
		return
	}

	setSessionCookie(w, token, h.svc.SessionTTL)
	web.WriteJSON(w, http.StatusOK, userResponse{
		ID: user.ID.String(), Email: user.Email, IsAdmin: user.IsAdmin,
	})
}

func (h *Handlers) Logout(w http.ResponseWriter, r *http.Request) {
	c, err := r.Cookie(cookieName)
	if err == nil {
		_ = h.svc.Logout(r.Context(), c.Value)
	}
	clearSessionCookie(w)
	w.WriteHeader(http.StatusNoContent)
}

func setSessionCookie(w http.ResponseWriter, token string, ttl time.Duration) {
	http.SetCookie(w, &http.Cookie{
		Name:     cookieName,
		Value:    token,
		Path:     "/",
		HttpOnly: true,
		Secure:   false, // proxied through nginx; flip to true behind TLS in production
		SameSite: http.SameSiteLaxMode,
		MaxAge:   int(ttl.Seconds()),
	})
}

func clearSessionCookie(w http.ResponseWriter) {
	http.SetCookie(w, &http.Cookie{
		Name: cookieName, Value: "", Path: "/", HttpOnly: true, MaxAge: -1,
	})
}

	})

	srv := &http.Server{
		Addr:         addr,
		Handler:      web.NewRouter(authH),
		ReadTimeout:  15 * time.Second,
		WriteTimeout: 30 * time.Second,
	}

	logger.Info("api listening", "addr", addr)
	if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("server stopped", "err", err)
		os.Exit(1)
	}
}

func getenv(k, fallback string) string {
	if v := os.Getenv(k); v != "" {

	}

export type User = { id: string; email: string; is_admin: boolean };

async function request<T>(method: string, path: string, body?: unknown): Promise<T> {
  const res = await fetch(path, {
    method,
    credentials: "include",
    headers: body ? { "Content-Type": "application/json" } : undefined,
    body: body ? JSON.stringify(body) : undefined,
  });
  if (!res.ok) {
    let msg = `request failed: ${res.status}`;
    try {
      const data = await res.json();
      if (data?.error) msg = data.error;
    } catch {
      // ignore JSON parse errors on error bodies
    }
    throw new Error(msg);
  }
  if (res.status === 204) return undefined as T;
  return res.json() as Promise<T>;
}

export const api = {
  register: (email: string, password: string) =>
    request<User>("POST", "/api/auth/register", { email, password }),
  login: (email: string, password: string) =>
    request<User>("POST", "/api/auth/login", { email, password }),
  logout: () => request<void>("POST", "/api/auth/logout"),
  me: () => request<User>("GET", "/api/auth/me"),
};
